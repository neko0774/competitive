#include<bits/stdc++.h>
using namespace std;

using ll = long long;

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int N;
  cin >> N;
  vector<ll> g(N + 1);
  g[0] = g[1] = 0;
  for(int x = 2; x <= N; x++) {
    vector<int> v;
    for(int y = 1; y <= x / 2; y++) {
      cout << x - y << " " << g[x - y] << endl;
      v.push_back(g[x - y]);
    }
    sort(v.begin(), v.end());
    v.erase(unique(v.begin(), v.end()), v.end());
    for(int i = 0; i < v.size(); i++) {
      cout << v[i] << " ";
    } cout << endl;
    for(int i = 0; i <= (int)v.size(); i++) {
      if(i == v.size() || v[i] != i) g[x] = i;
    }
  }

  for(int i = 1; i <= N; i++) {
    printf("%2d : %2lld\n", i, g[i]);
  }
}
